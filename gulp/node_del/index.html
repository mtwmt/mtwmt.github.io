gulp：del

刪除多餘的資料夾或文件

var gulp = require('gulp');
var del = require('del');

gulp.task('clean', function() {
  //目的資料夾
  del(['public/include/**', 'public/scss/**']);
});

終端機指令： npm gulp clean 或 yarn gulp clean
